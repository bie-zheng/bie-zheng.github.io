157        function getRandomPoem() {
158            let randomIndex = Math.floor(Math.random() * poemData.length);
159            currentPoem = poemData[randomIndex].poem;
160        }

162        function removePunctuation(text) {
163            // 移除所有标点符号
164            return text.replace(/[，。！？；：“”‘’《》【】、]/g, "");
165        }
166
167        function levenshteinDistance(a, b) {
168            if (a.length === 0) return b.length;
169            if (b.length === 0) return a.length;
170
171            let matrix = [];
172            for (let i = 0; i <= b.length; i++) {
173                matrix[i] = [i];
174            }
175            for (let j = 0; j <= a.length; j++) {
176                matrix[0][j] = j;
177            }
178
179            for (let i = 1; i <= b.length; i++) {
180                for (let j = 1; j <= a.length; j++) {
181                    if (b.charAt(i - 1) === a.charAt(j - 1)) {
182                        matrix[i][j] = matrix[i - 1][j - 1];
183                    } else {
184                        matrix[i][j] = Math.min(
185                            matrix[i - 1][j - 1] + 1,
186                            matrix[i][j - 1] + 1,
187                            matrix[i - 1][j] + 1
188                        );
189                    }
190                }
191            }
192
193            return matrix[b.length][a.length];
194        }
195
196        function findClosestPoem(guess) {
197            let closestPoem = "";
198            let minDistance = Infinity;
199
200            for (let poem of poemData) {
201                let distance = levenshteinDistance(guess, poem.poem);
202                if (distance < minDistance) {
203                    minDistance = distance;
204                    closestPoem = poem.poem;
205                }
206            }
207
208            return closestPoem;
209        }

211        function submitGuess() {
212            if (guessesLeft <= 0) {
213                document.getElementById("message").innerText = "猜测次数已用完，更换谜底。";
214                getRandomPoem();
215                guessesLeft = 20;
216                document.getElementById("guesses").innerHTML = "";
217                return;
218            }
219
220            let guess = document.getElementById("guessInput").value;
221            // 移除标点符号
222            guess = removePunctuation(guess);
223            // 截取前14个字
224            guess = guess.substring(0, 14);
225
226            if (guess.length < 14) {
227                alert("请输入至少14个有效汉字！");
228                return;
229            }
230
231            // 检查用户输入是否在词库中
232            let isInLibrary = poemData.some(poem => poem.poem === guess);
233            if (!isInLibrary) {
234                let closestPoem = findClosestPoem(guess);
235                document.getElementById("message").innerText = `您输入的古诗词不在词库中，最相近的词句是：${closestPoem}`;
236                return;
237            }
238
239            let guessDisplay = document.createElement("div");
240            guessDisplay.className = "guess";
241            for (let i = 0; i < 14; i++) {
242                let char = guess[i];
243                let span = document.createElement("span");
244                span.innerText = char;
245                if (char === currentPoem[i]) {
246                    span.className = "correct";
247                } else if (currentPoem.includes(char)) {
248                    span.className = "present";
249                } else {
250                    span.className = "absent";
251                }
252                guessDisplay.appendChild(span);
253            }
254            document.getElementById("guesses").appendChild(guessDisplay);
255
256            if (guess === currentPoem) {
257                document.getElementById("message").innerText = "恭喜你，猜对了！";
258                return;
259            }
260
261            guessesLeft--;
262            document.getElementById("message").innerText = `剩余猜测次数：${guessesLeft}`;
263            document.getElementById("guessInput").value = "";
264        }
